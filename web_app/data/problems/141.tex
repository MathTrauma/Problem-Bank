\numbering 삼각형 $ABC$의 $\overline{BC}$ 위의 임의의 한 점 $D$에 대하여 
삼각형 $ABC, ABD, ADC$의 외심을 각각 $O, O_1, O_1$이라 하면,  
네 점 $A, O, O_1, O_2$은 같은 원 위의 점임을 증명하시오. \\ 

\begin{center} \begin{tikzpicture}
	\draw[thick] (1, 3) coordinate (A) node[above] {$A$}
		-- (-2,0) coordinate (B) node[left] {$B$} 
		-- (2, 0) coordinate (C) node[right] {$C$} -- cycle;

	\coordinate (D) at ($(B)!0.6!(C)$);
	\node[thin, below] at (D) {$D$};
	\draw[dashed] (A)--(D);

	\draw let \p1=($(A)!0.5!(B)$), \p2=($(A)!0.5!(C)$), 
		\p3=($(\p1)!1!90:(B)$), \p4=($(\p2)!1!-90:(C)$),	
		\p5=(intersection of \p1--\p3 and \p2--\p4)    
		in coordinate (O) at (\p5) ;

	\draw[fill=red] (O) circle(1pt) node[right] {\footnotesize $O$};
	\node[thin, draw, circle through=(A)] at (O) {};

	%%%%%%%%%%%%%
	\draw let \p1=($(A)!0.5!(B)$), \p2=($(A)!0.5!(D)$), 
		\p3=($(\p1)!1!90:(B)$), \p4=($(\p2)!1!-90:(D)$),	
		\p5=(intersection of \p1--\p3 and \p2--\p4)    
		in coordinate (O1) at (\p5) ;

	\draw[fill=red] (O1) circle(1pt) node[left] {\footnotesize $O_1$};
	\node[thick, draw=red, circle through=(A)] at (O1) {};	

	%%%%%%%%%%%%%
	\draw let \p1=($(A)!0.5!(D)$), \p2=($(A)!0.5!(C)$), 
		\p3=($(\p1)!1!90:(D)$), \p4=($(\p2)!1!-90:(C)$),	
		\p5=(intersection of \p1--\p3 and \p2--\p4)    
		in coordinate (O2) at (\p5) ;

	\draw[fill=red] (O2) circle(1pt) node[right] {\footnotesize $O_2$};
	\node[thick, draw=blue, circle through=(A)] at (O2) {};	

\end{tikzpicture} \end{center}
