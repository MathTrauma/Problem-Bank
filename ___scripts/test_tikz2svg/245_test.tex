\documentclass[tikz,border=5pt]{standalone}
\usepackage{tkz-euclide}
\begin{document}

\begin{tikzpicture}[scale=0.5]

% 1. 점 A, B, C
\tkzDefPoint(-1,7){A}
\tkzDefPoint(-10,0){B}
\tkzDefPoint(10,0){C}

\tkzDrawPoints(A,B,C)
\tkzLabelPoints(A,B,C)

% 2. 무게중심 G
\tkzDefBarycentricPoint(A=1,B=1,C=1)
\tkzGetPoint{G}
\tkzDrawPoint(G)
\tkzLabelPoint[above](G){$G$}

% 3. A,C를 7:11로 내분하는 점 P
\tkzDefBarycentricPoint(A=11,C=7)    \tkzGetPoint{P}
\tkzDrawPoint(P)
\tkzLabelPoint[above right](P){$P$}

% 4. Q = PG ∩ BC, R = PG ∩ AB
\tkzInterLL(P,G)(B,C)
\tkzGetPoint{Q}
\tkzDrawPoint(Q)
\tkzLabelPoint[below](Q){$Q$}

\tkzInterLL(P,G)(A,B)
\tkzGetPoint{R}
\tkzDrawPoint(R)
\tkzLabelPoint[left](R){$R$}

% 5. 선분 AR, PR
\tkzDrawSegments(A,R P,R)

\end{tikzpicture}

\end{document}
